186 void get_wsv_group_ids(ArrayOfIndex& ids, String name) {
187  ids.resize(0);
188 
189  Index pos = 0;
190  while (pos < name.nelem()) {
191  switch (name[pos]) {
192  case ' ':
193  case '\r':
194  case '\t':
195  case '#':
196  name.erase(pos, 1);
197  break;
198  default:
199  pos++;
200  }
201  }
202 
203  pos = 0;
204  Index prev = 0;
205  while (pos < name.nelem()) {
206  while (pos < name.nelem() && name[pos] != ',') pos++;
207  Index id = get_wsv_group_id(name.substr(prev, pos - prev));
208  if (id == -1) {
209  ids.resize(0);
210  return;
211  }
212  ids.push_back(id);
213  pos++;
214  prev = pos;
215  }
216 }

232 String get_array_groups_as_string(bool basetype_is_group,
233  bool return_basetype_only) {
234  using global_data::wsv_group_names;
235  String arraygroups;
236 
237  bool first = true;
238  for (Index i = 0; i < wsv_group_names.nelem(); i++) {
239  if (wsv_group_names[i].substr(0, String("ArrayOf").length()) == "ArrayOf") {
240  const String basetype = wsv_group_names[i].substr(
241  String("ArrayOf").length(), wsv_group_names[i].length());
242  bool basetype_exists = (get_wsv_group_id(basetype) != -1);
243 
244  if (return_basetype_only) {
245  // Return only the basetype of the array,
246  // skip arrays whose basetype is not a WSV group
247  if (basetype_exists) {
248  if (!first)
249  arraygroups += ", ";
250  else
251  first = false;
252  arraygroups += basetype;
253  }
254  } else {
255  if (!basetype_is_group || (basetype_is_group && basetype_exists)) {
256  if (!first)
257  arraygroups += ", ";
258  else
259  first = false;
260  arraygroups += wsv_group_names[i];
261  }
262  }
263  }
264  }
265  return arraygroups;
266 }
