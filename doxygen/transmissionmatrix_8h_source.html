831  Index i) {

856  break;

859  R2[i].noalias() = Eigen::Vector2d(a.Kjj()[i], a.K12()[i]) * B[i] +
860  Eigen::Vector2d(S.Kjj()[i], S.K12()[i]);
861  else

863  break;
864  default:
865  if (not S.IsEmpty())
866  R1[i][0] = a.Kjj()[i] * B[i] + S.Kjj()[i];
867  else
