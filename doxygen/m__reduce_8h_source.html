40 inline Index num_elem_from_dim_sizes(const ArrayOfIndex& dim_sizes) {
41  Index m = 1;
42  for (ArrayOfIndex::const_iterator it = dim_sizes.begin();
43  it != dim_sizes.end();
44  it++)
45  m *= *it;
46 
47  return m;
48 }
